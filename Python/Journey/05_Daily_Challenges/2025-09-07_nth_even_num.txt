"""
Daily Challenge - 7th September 2025
------------------------------------
Challenge: Nth Even Number

Task:
Write a program that gets an integer n from input and outputs the nth even number starting from 0.

Examples:
Input: 1  → Output: 0   (0 is the first even number)
Input: 2  → Output: 2   (2 is the second even number)
Input: 100 → Output: 198
"""

# Read input
n = int(input())

# Calculate nth even number
nth_even = (n - 1) * 2

# Print result
print(nth_even)
